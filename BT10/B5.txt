    boolean checkBST(Node root) {
        List<Integer> list = new ArrayList<>();
        inOrderCheck(root, list);

        for(int i = 0; i < list.size()-1; i++) {
            if(list.get(i) >= list.get(i+1)) {
                return false;
            }
        }
        return true;
    }

    void inOrderCheck(Node root, List<Integer> list) {
        if(root == null) return;
        inOrderCheck(root.left, list);
        list.add(root.data);
        inOrderCheck(root.right, list);
    }
